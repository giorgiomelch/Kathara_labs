!  BGP CONFIGURATION

debug bgp keepalives
debug bgp updates in
debug bgp updates out

router bgp MODIFICA
    !!!disables the check of the existence of incoming and outgoing filters
	no bgp ebgp-requires-policy
    !!!this command modifies the network statement disabling the check of the existence of the prefix in the kernel forwarding table - used to announce prefixes that may not be local or may not even exist
	no bgp network import-check
    !
    ! NEIGHBORS
    !
    neighbor MODIFICA remote-as MODIFICA
    !
    ! NETWORKS
    !
    network MODIFICA/MODIFICA
    !
    ! POLICIES
    ! in=annuncia, out=accetta
    !neighbor MODIFICA default-originate !se isp upstream che segnala 0.0.0.0
    neighbor MODIFICA prefix-list asXIn in
    neighbor MODIFICA prefix-list asXOut out
    !
    ip prefix-list asXIn deny MODIFICA/MODIFICA
    ip prefix-list asXIn permit MODIFICA/MODIFICA
    ip prefix-list asXOut deny MODIFICA/MODIFICA
    ip prefix-list asXOut permit MODIFICA/MODIFICA
    ip prefix-list defaultRoute permit any
!
log file /var/log/frr/frr.log
